<!-- FIREBASE -->
<script type="module">
import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js";
import {
  getFirestore, collection, addDoc, getDocs
} from "https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js";
import {
  getStorage, ref, uploadBytes, getDownloadURL
} from "https://www.gstatic.com/firebasejs/10.7.1/firebase-storage.js";

/* 🔥 ВСТАВИШЬ СВОИ ДАННЫЕ СЮДА */
const firebaseConfig = {
  apiKey: "XXX",
  authDomain: "XXX",
  projectId: "XXX",
  storageBucket: "XXX",
};

const app = initializeApp(firebaseConfig);
const db = getFirestore(app);
const storage = getStorage(app);

let currentCategory = "";

/* ДОБАВИТЬ КАТЕГОРИЮ */
window.addCategory = async () => {
    const name = document.getElementById("categoryName").value;
    if (!name) return;

    await addDoc(collection(db, "categories"), { name });
    loadCategories();
};

/* ЗАГРУЗКА ФАЙЛОВ */
window.uploadFiles = async () => {
    const files = document.getElementById("fileUpload").files;

    for (let file of files) {
        const storageRef = ref(storage, currentCategory + "/" + file.name);

        await uploadBytes(storageRef, file);
        const url = await getDownloadURL(storageRef);

        await addDoc(collection(db, "images"), {
            category: currentCategory,
            url: url,
            type: file.type
        });
    }

    loadImages();
};

/* ЗАГРУЗКА КАТЕГОРИЙ */
async function loadCategories() {
    const snap = await getDocs(collection(db, "categories"));
    const container = document.getElementById("categories");
    container.innerHTML = "";

    snap.forEach(doc => {
        const cat = doc.data().name;

        const div = document.createElement("div");
        div.className = "category";
        div.innerText = cat;

        div.onclick = () => {
            currentCategory = cat;
            loadImages();
        };

        container.appendChild(div);
    });
}

/* ЗАГРУЗКА ФОТО */
async function loadImages() {
    const snap = await getDocs(collection(db, "images"));
    const container = document.getElementById("images");
    container.innerHTML = "";

    snap.forEach(doc => {
        const data = doc.data();
        if (data.category !== currentCategory) return;

        let el;

        if (data.type.startsWith("video")) {
            el = document.createElement("video");
            el.src = data.url;
        } else {
            el = document.createElement("img");
            el.src = data.url;
        }

        el.onclick = () => order(data.url);

        container.appendChild(el);
    });
}

/* ЗАКАЗ */
function order(url) {
    const text = "Хочу заказать с этим изображением: " + url;

    const telegram = "https://t.me/your_username?text=" + encodeURIComponent(text);

    window.open(telegram, "_blank");
}

loadCategories();
